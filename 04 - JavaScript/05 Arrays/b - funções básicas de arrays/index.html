push: adiciona um elemento no final do array

      let hitchedSpaceships = ["Colossus", "Elemental", "Helmet"];
      console.log(hitchedSpaceships);
      //(3) ['Colossus', 'Elemental', 'Helmet']

      hitchedSpaceships.push("Supernova");
      console.log(hitchedSpaceships);
      //(4) ['Colossus', 'Elemental', 'Helmet', 'Supernova']

Pop: remove o elemento da última posição e, em uma variável, retorna esse último elemento removido. 

          let removedSpaceship = hitchedSpaceships.pop();
          console.log(removedSpaceship);

Shift: remove o elemento que está na primeira posição. E, assim como a pop, também retorna o elemento que foi removido. 

          let removedWithShift = hitchedSpaceships.shift();
          console.log(removedWithShift); //Colossus
          console.log(hitchedSpaceships); //(2) ['Elemental', 'Helmet']

Unshift: oposto da shift, adicionando um elemento ao começo do array.

          let addUnshift = hitchedSpaceships.unshift("Phoenix");
          console.log(hitchedSpaceships);
          //(3) ['Phoenix', 'Elemental', 'Helmet']

IndexOf: retorna a posição de um elemento e retorna também resposta quando não encontra (-1). 

          let hitchedSpaceships = ["Colossus", "Elemental", "Helmet"];
          console.log(hitchedSpaceships);

          hitchedSpaceships.push("Supernova");
          console.log(hitchedSpaceships);

          let removedSpaceship = hitchedSpaceships.pop();
          console.log(removedSpaceship);

          let removedWithShift = hitchedSpaceships.shift();
          console.log(removedWithShift);
          console.log(hitchedSpaceships);

          let addUnshift = hitchedSpaceships.unshift("Phoenix");
          console.log(hitchedSpaceships);

          console.log(hitchedSpaceships.length);

          let elementalPosition = hitchedSpaceships.indexOf("Elemental");
          console.log(elementalPosition);
